def by_title_ascending (book_a, book_b) :
    title_a = book_a['lower_title']
    title_b = book_b['lower_title']
    if len(title_a) == 0 :
        return True
    elif len(title_b) == 0 :
        return False
    first_letter_a = title_a[0]
    first_letter_b = title_b[0]
    if ord(first_letter_a) == ord(first_letter_b) :
        book_a['title_lower'].pop(0)
        book_b['title_lower'].pop(0)
        return by_title_ascending(book_a, book_b)
    elif ord(first_letter_a) > ord(first_letter_b) :
        return True
    else :
        return True
